** 1.2: Dimensional combinations
#+ATTR_REVEAL: :frag (roll-in)
#+ATTR_HTML: :style text-align:left

#+ATTR_REVEAL: :frag (roll-in)
#+ATTR_HTML: :style text-align:left

#+ATTR_HTML: :style text-align:left
